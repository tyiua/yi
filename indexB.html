        int cifang(int n,int k)
        {
            if(k == 1)
                return n;
            else { 
                return cifang(n,k-1)*n;
            }
        }
        int main (){
            int n,k;
            scanf("%d %d",&n,&k);
            printf("%d",cifang(n,k));
            return 0;
        }
